create schema order_system;

create table order_system.menus(
    menu_id uuid not null default gen_random_uuid(),
    menu_classification varchar not null,
    menu_name varchar not null,
    menu_price integer not null,
    sale_flg boolean not null,
    primary key (menu_id)
);

create table order_system.seats(
    seat_id integer not null,
    vacant_seat_flg boolean not null default true,
    number_of_people integer not null default 0,
    number_of_seats integer not null,
    primary key (seat_id)
);

create table order_system.customers(
    customer_id uuid not null default gen_random_uuid(),
    entry_time timestamp not null DEFAULT CURRENT_TIMESTAMP,
    number_of_people integer not null,
    seat_id integer not null,
    exit_time timestamp,
    payment_price integer not null default 0,
    payment_method integer,
    check_time timestamp,
    primary key (customer_id)
);

create table order_system.orders(
    order_time timestamp not null,
    customer_id uuid not null,
    menu_id uuid not null,
    number_of_orders integer not null,
    seat_id integer not null,
    distributed_flg boolean not null,
    received_flg boolean not null,
    foreign key (customer_id) references order_system.customers(customer_id)
);

CREATE OR REPLACE FUNCTION order_system.entry_customer_func()
RETURNS TRIGGER AS $$
DECLARE
    arg1 INT;
    arg2 INT;
BEGIN
    arg1 := NEW.number_of_people; 
    arg2 := NEW.seat_id; 

    insert into order_system.customers (number_of_people, seat_id) values (arg1, arg2);
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER entry_customer_trig
AFTER UPDATE OF vacant_seat_flg ON order_system.seats
FOR EACH ROW
WHEN (OLD.vacant_seat_flg = true AND NEW.vacant_seat_flg = false)
EXECUTE FUNCTION order_system.entry_customer_func();

create schema hoge;

create table hoge.tbl_a(
    id integer not null,
    name varchar not null,
    flg boolean not null,
    primary key (id)
);

create table hoge.tbl_b(
    id uuid not null default gen_random_uuid(),
    datetime timestamp not null DEFAULT CURRENT_TIMESTAMP,
    num integer not null,
    name varchar not null,
    primary key (id)
);

CREATE OR REPLACE FUNCTION hoge.func_a()
RETURNS TRIGGER AS $$
DECLARE
    arg1 INT;
    arg2 TEXT;
BEGIN
    arg1 := NEW.id;  -- NEWレコードのidカラムの値をarg1に代入
    arg2 := NEW.name;  -- NEWレコードのnameカラムの値をarg2に代入

    INSERT INTO hoge.tbl_b (num, name) VALUES (arg1, arg2);
    RETURN NEW;  -- トリガー関数がトリガーによって実行された行の状態を返す
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trig_a
AFTER UPDATE OF flg ON hoge.tbl_a
FOR EACH ROW
WHEN (OLD.flg = true AND NEW.flg = false)
EXECUTE FUNCTION hoge.func_a();  -- 引数を指定せずにトリガー関数を呼び出す

insert into hoge.tbl_a (id, name, flg) values
(1, 'a', true),
(2, 'b', true),
(3, 'c', true),
(4, 'd', true),
(5, 'e', true);

select * from hoge.tbl_a
ORDER BY id ASC;

select * from hoge.tbl_b
ORDER BY datetime ASC;

update hoge.tbl_a SET flg = false where id = 3;
